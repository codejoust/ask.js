Ask.js - the easiest way to ask someone something.

Usage:

var ask = require('ask');
var ask_req = ask.to('dinner').at('Starbucks').near('Chattanooga, TN').around('Friday Night');
ask_req.on('email_back', function(req, res){ res.send('Thanks!'); });
ask_req.send_by_email('participat@therealmeaningof.com');
ask_req.participants.push(new ask.user('asker@therealmeaningof.com'))

Why:

Because planning real face-to-face stuff can be so hard.
